a. Scenario: A horse racing meeting which has races and runners in the races. With each race as the key, which would be a String labelled Race1, Race2 etc., and the values would be a Set of Strings where each String represents a runner in the race. The types I have chosen are suitable because the String as the key as it is just the name of the race as a word, and a Set of Strings as the runners as they are a collection of the names of the horses. (90 words)

b. private Map String, Set String >> races;
/**
* Constructor for objects of class RaceMeeting
*/
public RaceMeeting()
{
races = new TreeMap();
}
/**
* Creates a sample race meeting
*
*/
public void createMeeting()
{
Set String runners = new HashSet();
runners.add("Red Rum");
runners.add("Arkle");
runners.add("Kauto Star");
races.put("Race1", runners);
runners = new HashSet();
runners.add("Frankel");
runners.add("See The Stars");
runners.add("New Approach");
races.put("Race2", runners);
}

c. /**
* Iterates over the map to print out each race with the runners in the race
* @param takes no arguments
*/
public void printMap()
{
String output;
SetString runners;
for (String eachRace : races.keySet())
{
output = eachRace;
runners = races.get(eachRace);
System.out.println(output);
System.out.println(runners);
}
}
/**
* Prints the value of the key if it exists otherwise prints
* that the key is not present
* @param takes a String argument
*/
public void printMapValue(String aKey)
{
if (races.get(aKey) != null)
{
System.out.println(races.get(aKey));
}
else
{
System.out.println("The key is not present");
}
}
/**
* Adds the key and value to the map from the given arguments,
* @param takes two String arguments separate by a ,
*/
public void addMapEntry(String aString, String aSet)
{
SetString runners = new HashSet();
runners.add(aSet);
races.put(aString, runners);
}

d. /**
* If the String argument exists in the map then this is removed
* @param takes a String argument
* @return true if it has been removed, false if not
*/
public boolean deleteEntry(String aString)
{
if (races.containsKey(aString))
{
races.remove(aString);
return true;
}
else
{
return false;
}
}
/**
* Adds a new value to a collection of values of a particular key
* @param takes two String arguments
* @return the new set
*/
public Set String addValue(String aKey, String aValue)
{
Set String newString = races.get(aKey);
newString.add(aValue);
races.put(aKey, newString);
return races.get(aKey);
}
/**
* Deletes a value from a collection of values for a particular key
* @param takes two String arguments
* @return the new set
*/
public Set String deleteValue(String aKey, String aValue)
{
Set String newString = races.get(aKey);
newString.remove(aValue);
races.put(aKey, newString);
return races.get(aKey);
}
